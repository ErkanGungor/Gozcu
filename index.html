            <h1 class="text-3xl font-bold text-gray-800">Gözcü Projesi</h1>

            <label for="message" class="block text-sm font-medium text-gray-700">Analiz edilecek mesajınızı buraya yazın:</label>

            <textarea id="message" rows="6" class="mt-2 w-full p-3 border border-gray-300 rounded-lg shadow-sm focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition" placeholder="Örn: Arkadaşlar, rakip firma ile görüştüm. Fiyatları artıralım diyorum..."></textarea>

        </div>


        <!-- Buton Alanı -->

        <div class="mt-6">

            <button id="analyzeButton" class="w-full bg-blue-600 text-white font-bold py-3 px-4 rounded-lg hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500 transition-transform transform hover:scale-105">

                Mesajı Analiz Et

            </button>

        </div>


        <!-- Sonuç Gösterim Alanı -->

        <div id="resultContainer" class="mt-8 p-6 rounded-lg hidden">

            <h3 id="resultTitle" class="text-lg font-bold"></h3>

            <p id="resultScore" class="mt-1 text-sm"></p>

            <p id="resultDescription" class="mt-2"></p>

        </div>

    <script>

        const analyzeButton = document.getElementById('analyzeButton');

        const messageInput = document.getElementById('message');

        const resultContainer = document.getElementById('resultContainer');

        const resultTitle = document.getElementById('resultTitle');

        const resultScore = document.getElementById('resultScore');

        const resultDescription = document.getElementById('resultDescription');


        /**

         * Colab'daki analiz fonksiyonunun aynısı.

         * Bir metni analiz edip potansiyel rekabet hukuku risklerini değerlendirir.

         */

        function riskAnalizi(metin) {

            if (!metin.trim()) {

                return {

                    sonuc: "Boş Metin",

                    risk_skoru: 0.0,

                    aciklama: "Lütfen analiz edilecek bir metin girin."

                };

            }


            const temiz_metin = metin.toLowerCase();

            const risk_katalogu = {

                "Fiyat Anlaşması Riski": {

                    "kelimeler": ["fiyatları artıralım", "fiyatları yükseltelim", "fiyat sabitleyelim", "aynı fiyattan", "fiyat listesi", "zam yapalım"],

                    "skor": 0.95

                },

                "Pazar Paylaşımı Riski": {

                    "kelimeler": ["bölge paylaşımı", "pazarı paylaşalım", "sen o bölgeye girme", "müşterileri paylaşalım", "sen o müşteriye gitme"],

                    "skor": 0.90

                },

                "Rakip Bilgisi Riski": {

                    "kelimeler": ["rakibin teklifi", "rakip fiyatı", "rakibin maliyeti", "gizli bilgi"],

                    "skor": 0.75

                }

            };


            let tespit_edilen_riskler = [];

            for (const [kategori, detay] of Object.entries(risk_katalogu)) {

                for (const kelime of detay.kelimeler) {

                    if (temiz_metin.includes(kelime)) {

                        tespit_edilen_riskler.push({

                            "risk_kategorisi": kategori,

                            "risk_skoru": detay.skor,

                            "tetikleyen_ifade": kelime

                        });

                    }

                }

            }


            if (tespit_edilen_riskler.length === 0) {

                return {

                    sonuc: "Temiz",

                    risk_skoru: 0.10,

                    aciklama: "Metinde belirgin bir rekabet hukuku riski tespit edilmedi."

                };

            } else {

                let en_yuksek_risk = tespit_edilen_riskler.sort((a, b) => b.risk_skoru - a.risk_skoru)[0];

                en_yuksek_risk.sonuc = 'Riskli';

                en_yuksek_risk.aciklama = `'${en_yuksek_risk.tetikleyen_ifade}' ifadesi nedeniyle yüksek risk tespit edildi.`;

                return en_yuksek_risk;

            }

        }


        // Butona tıklandığında çalışacak fonksiyon

        analyzeButton.addEventListener('click', () => {

            const mesaj = messageInput.value;

            const analizSonucu = riskAnalizi(mesaj);


            // Önceki stil sınıflarını temizle

            resultContainer.classList.remove('bg-red-100', 'border-red-500', 'text-red-900', 'bg-green-100', 'border-green-500', 'text-green-900', 'bg-yellow-100', 'border-yellow-500', 'text-yellow-900', 'hidden');

            resultContainer.classList.add('border');



            if (analizSonucu.sonuc === 'Riskli') {

                resultContainer.classList.add('bg-red-100', 'border-red-500', 'text-red-900');

                resultTitle.textContent = `ANALİZ SONUCU: RİSKLİ`;

                resultScore.textContent = `Risk Kategorisi: ${analizSonucu.risk_kategorisi} | Risk Skoru: ${analizSonucu.risk_skoru.toFixed(2)}`;

            } else if (analizSonucu.sonuc === 'Temiz') {

                resultContainer.classList.add('bg-green-100', 'border-green-500', 'text-green-900');

                resultTitle.textContent = 'ANALİZ SONUCU: TEMİZ';

                resultScore.textContent = `Risk Skoru: ${analizSonucu.risk_skoru.toFixed(2)}`;

            } else { // Boş metin durumu

                resultContainer.classList.add('bg-yellow-100', 'border-yellow-500', 'text-yellow-900');

                resultTitle.textContent = 'UYARI';

                resultScore.textContent = '';

            }


            resultDescription.textContent = analizSonucu.aciklama;

            resultContainer.classList.remove('hidden');

        });


    </script>
